server {
    listen 8080;
    host 0.0.0.0;
    server_name localhost;
    root www/localhost;
    
    error 404 404.html;
    client_max_body_size 1M;

    location {
        path /;
        index index.html;
        listing true;
        method GET,POST;
    }
}
